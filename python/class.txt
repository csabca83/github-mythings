class Car:
	def __init__(self,marka,ar):
		self.marka=marka
		self.ar=ar
		self.km=100000  --- default value
	def osszegzes(self):
		print('marka',self.marka)
		print('ar',self.ar)
		print('km',self.km)
	def updater(self,uj_km):
		self.km=uj_km   --- ezzel valtoztatom meg a default valuer
fabia=Car(marka='skoda',ar=1500)
fabia.updater(15)
fabia.osszegzes()
